Функции для переиспользования кода

Хороший автоматизационный скрипт — это набор небольших функций, каждая из которых делает одно конкретное дело. Это позволяет легко тестировать, изменять и комбинировать логику.

PYTHON 
import os
import logging
from pathlib import Path
from typing import Optional

# ✅ Принцип: одна функция — одна задача

def ensure_dir(path: str) -> Path:
    """Создаёт директорию если не существует. Возвращает Path."""
    dir_path = Path(path)
    dir_path.mkdir(parents=True, exist_ok=True)
    logging.info(f"Директория готова: {dir_path}")
    return dir_path


def get_files_by_ext(folder: str, ext: str) -> list:
    """Возвращает список файлов с указанным расширением."""
    folder_path = Path(folder)
    if not folder_path.exists():
        logging.warning(f"Папка не найдена: {folder}")
        return []

    files = list(folder_path.glob(f"*.{ext.lstrip('.')}"))
    logging.info(f"Найдено {len(files)} файлов .{ext} в {folder}")
    return files


def safe_move(src: str, dst: str) -> bool:
    """Безопасно перемещает файл. Возвращает True при успехе."""
    try:
        src_path = Path(src)
        dst_path = Path(dst)

        if not src_path.exists():
            raise FileNotFoundError(f"Файл не найден: {src}")

        dst_path.parent.mkdir(parents=True, exist_ok=True)
        src_path.rename(dst_path)
        logging.info(f"✓ Перемещён: {src} → {dst}")
        return True

    except Exception as e:
        logging.error(f"✗ Ошибка перемещения {src}: {e}")
        return False


# Использование:
output_dir = ensure_dir("D:/Reports/2024/January")
csv_files  = get_files_by_ext("C:/Downloads", "csv")

for f in csv_files:
    safe_move(str(f), str(output_dir / f.name))

Логирование — твой лучший друг в автоматизации

Скрипт будет работать ночью, в выходные, без тебя рядом. Без логов ты никогда не узнаешь, что пошло не так. Logging — обязательный элемент любого production-скрипта.

PYTHON — logging_setup.py 
import logging
import sys
from pathlib import Path
from datetime import datetime


def setup_logging(log_dir: str = "logs", level: str = "INFO") -> None:
    """
    Настраивает логирование в файл И консоль одновременно.
    Файл: logs/script_2024-01-15.log
    """
    log_path = Path(log_dir)
    log_path.mkdir(exist_ok=True)

    log_file = log_path / f"script_{datetime.now().strftime('%Y-%m-%d')}.log"

    # Формат: [ВРЕМЯ] [УРОВЕНЬ] [ФАЙЛ:СТРОКА] — СООБЩЕНИЕ
    fmt = "%(asctime)s [%(levelname)8s] %(filename)s:%(lineno)d — %(message)s"

    logging.basicConfig(
        level=getattr(logging, level.upper()),
        format=fmt,
        datefmt="%H:%M:%S",
        handlers=[
            logging.FileHandler(log_file, encoding="utf-8"),
            logging.StreamHandler(sys.stdout),
        ],
    )
    logging.info(f"📝 Логирование запущено. Файл: {log_file}")


# В main.py — первая строка после импортов:
if __name__ == "__main__":
    setup_logging(level="DEBUG")

    logging.debug("Детальная отладочная информация")
    logging.info("✅ Скрипт запущен успешно")
    logging.warning("⚠️ Папка почти заполнена (85%)")
    logging.error("❌ Не удалось подключиться к API")
    logging.critical("💀 База данных недоступна, останов")
⚠️
Типичная ошибка: использовать print() вместо logging в production-скриптах. print не пишет в файл, не добавляет временную метку и не имеет уровней важности. Всегда используй logging.
🧠 ЧТО ВАЖНО ЗАПОМНИТЬ
  • Каждый проект — отдельный venv. Никаких исключений.
  • Всегда requirements.txt в git-репозитории.
  • Функции — маленькие, с docstring и аннотациями типов.
  • logging вместо print в любом скрипте, который работает автономно.
  • Секреты (токены, пароли) — только через .env, никогда в коде.
  • Структура проекта: src/, logs/, data/, tests/ — до написания первой строки.

Автоматическая сортировка файлов

Классический кейс: папка "Загрузки" захламлена сотнями файлов разных типов. Скрипт за секунды раскладывает их по папкам.

PYTHON — file_sorter.py 
import shutil
import logging
from pathlib import Path
from datetime import datetime

# Маппинг расширений → папки
FILE_CATEGORIES: dict[str, list[str]] = {
    "📄 Документы":  [".pdf", ".doc", ".docx", ".txt", ".odt"],
    "📊 Таблицы":    [".xls", ".xlsx", ".csv", ".ods"],
    "🖼️ Изображения":[".jpg", ".jpeg", ".png", ".gif", ".webp"],
    "🎬 Видео":      [".mp4", ".mkv", ".avi", ".mov"],
    "🎵 Музыка":     [".mp3", ".flac", ".wav", ".ogg"],
    "📦 Архивы":     [".zip", ".rar", ".7z", ".tar", ".gz"],
    "💻 Программы":  [".exe", ".msi", ".dmg", ".deb"],
    "🐍 Код":        [".py", ".js", ".html", ".css", ".json"],
}

def sort_downloads(source_dir: str) -> dict:
    """
    Сортирует файлы из source_dir по категориям.
    Возвращает статистику: {'📄 Документы': 5, ...}
    """
    source = Path(source_dir)
    stats  = {}
    errors = 0

    # Строим обратный маппинг: .pdf → '📄 Документы'
    ext_map: dict[str, str] = {}
    for category, exts in FILE_CATEGORIES.items():
        for ext in exts:
            ext_map[ext] = category

    for file in source.iterdir():
        if not file.is_file() or file.name.startswith("."):
            continue

        category = ext_map.get(file.suffix.lower(), "❓ Прочее")
        dest_dir = source / category
        dest_dir.mkdir(exist_ok=True)

        dest_file = dest_dir / file.name
        if dest_file.exists():   # если файл уже есть — добавляем timestamp
            ts = datetime.now().strftime("%H%M%S")
            dest_file = dest_dir / f"{file.stem}_{ts}{file.suffix}"

        try:
            shutil.move(str(file), str(dest_file))
            stats[category] = stats.get(category, 0) + 1
            logging.info(f"✓ {file.name} → {category}")
        except Exception as e:
            logging.error(f"✗ {file.name}: {e}")
            errors += 1

    logging.info(f"Готово. Перемещено: {sum(stats.values())}, ошибок: {errors}")
    return stats


# Запуск:
if __name__ == "__main__":
    result = sort_downloads("C:/Users/user/Downloads")
    for cat, count in result.items():
        print(f"  {cat}: {count} файл(ов)")

    observer = Observer()
    observer.schedule(AutoProcessor(), path, recursive=False)
    observer.start()

Архивация с логом выполнения

PYTHON — backup.py 
import zipfile, shutil, logging
from pathlib import Path
from datetime import datetime


def create_backup(source: str, dest_dir: str,
                   max_backups: int = 7) -> str:
    """
    Создаёт ZIP-архив папки с датой в имени.
    Автоматически удаляет старые архивы (хранит max_backups).
    """
    timestamp = datetime.now().strftime("%Y-%m-%d_%H-%M")
    source_path = Path(source)
    dest_path   = Path(dest_dir)
    dest_path.mkdir(parents=True, exist_ok=True)

    archive_name = dest_path / f"backup_{source_path.name}_{timestamp}.zip"

    # Создаём архив
    with zipfile.ZipFile(archive_name, "w", zipfile.ZIP_DEFLATED) as zf:
        for file in source_path.rglob("*"):
            if file.is_file():
                zf.write(file, file.relative_to(source_path))

    size_mb = archive_name.stat().st_size / 1024 / 1024
    logging.info(f"✅ Бэкап создан: {archive_name.name} ({size_mb:.1f} MB)")

    # Удаляем старые бэкапы
    backups = sorted(dest_path.glob("backup_*.zip"),
                    key=lambda x: x.stat().st_mtime)
    while len(backups) > max_backups:
        old = backups.pop(0)
        old.unlink()
        logging.info(f"🗑️ Удалён старый бэкап: {old.name}")

    return str(archive_name)


# Пример: ежедневный бэкап документов
create_backup(
    source="C:/Users/user/Documents",
    dest_dir="D:/Backups",
    max_backups=14   # хранить 2 недели
)
🧠 ЧТО ВАЖНО ЗАПОМНИТЬ
  • pathlib.Path — современный стандарт, используй вместо os.path.
  • shutil.move() — для перемещения, shutil.copy2() — для копирования с метаданными.
  • Всегда проверяй exist_ok=True при создании папок.
  • watchdog — лучший инструмент для мониторинга директорий в реальном времени.
  • Обрабатывай конфликты имён файлов — добавляй timestamp к дублям.

# pip install openpyxl
import openpyxl
from openpyxl.styles import Font, PatternFill, Alignment, Border, Side
from openpyxl.utils import get_column_letter
from openpyxl.chart import BarChart, Reference
from datetime import date


def create_sales_report(data: list[dict], output_path: str) -> None:
    """
    Генерирует красивый Excel-отчёт по продажам.
    data: [{'Менеджер': 'Иван', 'Продажи': 150000, 'Цель': 120000}]
    """
    wb = openpyxl.Workbook()
    ws = wb.active
    ws.title = "Продажи"

    # ── Стили ──────────────────────────────────────
    HEADER_FONT  = Font(bold=True, color="FFFFFF", size=12)
    HEADER_FILL  = PatternFill("solid", fgColor="1F4E79")
    ALT_FILL     = PatternFill("solid", fgColor="DEEAF1")
    CENTER       = Alignment(horizontal="center", vertical="center")
    thin_border  = Side(style="thin", color="AAAAAA")
    BORDER       = Border(
        left=thin_border, right=thin_border,
        top=thin_border, bottom=thin_border
    )

    # ── Заголовок отчёта ───────────────────────────
    ws.merge_cells("A1:E1")
    ws["A1"] = f"Отчёт по продажам — {date.today().strftime('%d.%m.%Y')}"
    ws["A1"].font      = Font(bold=True, size=16, color="1F4E79")
    ws["A1"].alignment = CENTER

    # ── Заголовки колонок ──────────────────────────
    headers = ["Менеджер", "Продажи (руб.)", "Цель (руб.)", "Выполнение %", "Статус"]
    for col, header in enumerate(headers, start=1):
        cell           = ws.cell(row=2, column=col, value=header)
        cell.font      = HEADER_FONT
        cell.fill      = HEADER_FILL
        cell.alignment = CENTER
        cell.border    = BORDER

    # ── Данные ─────────────────────────────────────
    for row_idx, row_data in enumerate(data, start=3):
        pct    = row_data["Продажи"] / row_data["Цель"] * 100
        status = "✅ Выполнен" if pct >= 100 else "⚠️ В работе"
        values = [row_data["Менеджер"], row_data["Продажи"],
                  row_data["Цель"], round(pct, 1), status]

        fill = ALT_FILL if row_idx % 2 == 0 else None
        for col, val in enumerate(values, start=1):
            cell = ws.cell(row=row_idx, column=col, value=val)
            if fill: cell.fill = fill
            cell.border    = BORDER
            cell.alignment = CENTER

    # ── Авторазмер колонок ─────────────────────────
    for col in ws.columns:
        max_len = max(len(str(cell.value or "")) for cell in col)
        ws.column_dimensions[get_column_letter(col[0].column)].width = max_len + 4

    wb.save(output_path)
    print(f"✅ Отчёт сохранён: {output_path}")

# Чистка данных

df["amount"] = pd.to_numeric(df["amount"], errors="coerce")
df = df[df["status"] != "cancelled"]   # исключить отменённые

# Группировка: продажи по менеджерам за текущий месяц
current_month = pd.Timestamp.now().month
monthly = df[df["date"].dt.month == current_month]

summary = monthly.groupby("manager").agg(
    total_sales = ("amount", "sum"),
    deal_count  = ("amount", "count"),
    avg_deal    = ("amount", "mean"),
).sort_values("total_sales", ascending=False)

summary["total_sales"] = summary["total_sales"].round(2)
summary["avg_deal"]    = summary["avg_deal"].round(2)

# Экспорт в Excel с несколькими листами
with pd.ExcelWriter("monthly_report.xlsx", engine="openpyxl") as writer:
    summary.to_excel(writer, sheet_name="Сводка")
    monthly.to_excel(writer, sheet_name="Детальные данные", index=False)

print(f"📊 Отчёт готов. Топ продавец: {summary.index[0]}")
print(summary.to_string())

requests + BeautifulSoup — базовый парсинг

PYTHON — parser_basic.py 
# pip install requests beautifulsoup4 lxml
import requests
from bs4 import BeautifulSoup
import csv, time, logging
from dataclasses import dataclass

@dataclass
class Product:
    name:  str
    price: float
    url:   str
    in_stock: bool


class ShopParser:
    def __init__(self, base_url: str):
        self.base_url = base_url
        self.session  = requests.Session()
        self.session.headers.update({
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
                          "AppleWebKit/537.36 Chrome/120.0.0.0 Safari/537.36",
            "Accept-Language": "ru-RU,ru;q=0.9,en;q=0.8",
        })

    def get_page(self, url: str, retries: int = 3) -> BeautifulSoup | None:
        """Загружает страницу с retry-логикой."""
        for attempt in range(retries):
            try:
                resp = self.session.get(url, timeout=10)
                resp.raise_for_status()
                return BeautifulSoup(resp.text, "lxml")
            except requests.RequestException as e:
                logging.warning(f"Попытка {attempt+1}/{retries} не удалась: {e}")
                time.sleep(2 ** attempt)  # экспоненциальный backoff
        return None

    def parse_catalog(self, category_url: str) -> list[Product]:
        """Парсит каталог товаров. Адаптируй CSS-селекторы под сайт."""
        products = []
        soup = self.get_page(category_url)
        if not soup: return products

        for card in soup.select(".product-card"):  # CSS-селектор карточки
            try:
                name_el  = card.select_one(".product-title")
                price_el = card.select_one(".price")
                link_el  = card.select_one("a[href]")

                if not all([name_el, price_el]):
                    continue

                price_text = price_el.text.replace(" ", "").replace("руб.", "").strip()

                products.append(Product(
                    name     = name_el.text.strip(),
                    price    = float(price_text),
                    url      = self.base_url + link_el["href"],
                    in_stock = "out-of-stock" not in card.get("class", []),
                ))
            except (ValueError, AttributeError) as e:
                logging.debug(f"Пропускаем карточку: {e}")

        time.sleep(1.5)  # вежливая задержка между запросами!
        return products

    def save_to_csv(self, products: list[Product], path: str) -> None:
        with open(path, "w", newline="", encoding="utf-8-sig") as f:
            writer = csv.writer(f)
            writer.writerow(["Название", "Цена", "URL", "В наличии"])
            for p in products:
                writer.writerow([p.name, p.price, p.url, p.in_stock])
        logging.info(f"✅ Сохранено {len(products)} товаров в {path}")

REST API — профессиональная интеграция

PYTHON — api_client.py 
import requests, os, logging
from dotenv import load_dotenv

load_dotenv()  # читаем .env файл

class APIClient:
    """Базовый класс для работы с REST API."""

    def __init__(self, base_url: str, api_key: str):
        self.base_url = base_url.rstrip("/")
        self.session  = requests.Session()
        self.session.headers.update({
            "Authorization": f"Bearer {api_key}",
            "Content-Type":  "application/json",
        })

    def get(self, endpoint: str, params: dict = None) -> dict:
        url  = f"{self.base_url}/{endpoint.lstrip('/')}"
        resp = self.session.get(url, params=params, timeout=15)
        resp.raise_for_status()
        return resp.json()

    def post(self, endpoint: str, data: dict) -> dict:
        url  = f"{self.base_url}/{endpoint.lstrip('/')}"
        resp = self.session.post(url, json=data, timeout=15)
        resp.raise_for_status()
        return resp.json()


# Пример: работа с API обменных курсов
API_KEY  = os.getenv("EXCHANGE_API_KEY")
client   = APIClient("https://api.exchangerate.host", API_KEY)

data     = client.get("latest", params={"base": "USD", "symbols": "RUB,EUR,CNY"})
usd_rub  = data["rates"]["RUB"]
logging.info(f"Курс USD/RUB: {usd_rub}")
🚫
Важно о парсинге: перед парсингом любого сайта проверяй robots.txt и условия использования. Никогда не делай более 1 запроса в секунду без явного разрешения. Парсинг для личного использования и анализа обычно законен; коммерческое использование данных без разрешения может нарушать ToS.
🧠 ЧТО ВАЖНО ЗАПОМНИТЬ
  • Всегда устанавливай User-Agent, иначе тебя заблокируют первым же запросом.
  • Retry-логика с экспоненциальным backoff — обязательна для надёжного парсера.
  • API-ключи только через .env, никогда в коде.
  • Задержка 1-3 секунды между запросами — уважение к серверу и защита от бана.
  • dataclasses — лучший способ хранить структурированные данные из парсера.

Email автоматизация

PYTHON — email_sender.py 
import smtplib, os, logging
from email.mime.multipart import MIMEMultipart
from email.mime.text import MIMEText
from email.mime.base import MIMEBase
from email import encoders
from pathlib import Path


def send_report_email(
    to_emails:  list[str],
    subject:    str,
    body_html:  str,
    attachments:list[str] = None,
) -> bool:
    """Отправляет HTML-письмо с вложениями через Gmail/SMTP."""
    smtp_host = os.getenv("SMTP_HOST", "smtp.gmail.com")
    smtp_port = int(os.getenv("SMTP_PORT", "587"))
    smtp_user = os.getenv("SMTP_USER")
    smtp_pass = os.getenv("SMTP_PASS")

    msg = MIMEMultipart("alternative")
    msg["From"]    = smtp_user
    msg["To"]      = ", ".join(to_emails)
    msg["Subject"] = subject
    msg.attach(MIMEText(body_html, "html", "utf-8"))

    # Добавляем вложения
    for file_path in (attachments or []):
        path = Path(file_path)
        with open(path, "rb") as f:
            part = MIMEBase("application", "octet-stream")
            part.set_payload(f.read())
        encoders.encode_base64(part)
        part.add_header("Content-Disposition", f'attachment; filename="{path.name}"')
        msg.attach(part)

    try:
        with smtplib.SMTP(smtp_host, smtp_port) as server:
            server.starttls()
            server.login(smtp_user, smtp_pass)
            server.sendmail(smtp_user, to_emails, msg.as_string())
        logging.info(f"✉️ Письмо отправлено: {', '.join(to_emails)}")
        return True
    except Exception as e:
        logging.error(f"❌ Ошибка отправки email: {e}")
        return False
